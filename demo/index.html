Floating Action Button


import { FloatingActionButton } from 'techui-react-components';

function App() {
  return (
    <FloatingActionButton
      icon="chat"
      color="primary"
      position="bottom-right"
      onClick={() => alert('Button clicked!')}
    />
  );
}

Interactive Testimonial Carousel

}
